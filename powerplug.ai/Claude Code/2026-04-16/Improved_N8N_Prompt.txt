<script>
(function(){
  var btn=document.querySelector('.sa-article .sa-float-top');
  if(!btn)return;
  var t=document.querySelector('.sa-article')||document.documentElement;
  if(!t.id)t.id='sa-top';
  btn.setAttribute('href','#'+t.id);
  function tog(){if((window.pageYOffset||document.documentElement.scrollTop)>300){btn.classList.add('show');}else{btn.classList.remove('show');}}
  btn.addEventListener('click',function(e){e.preventDefault();window.scrollTo({top:0,behavior:'smooth'});});
  window.addEventListener('scroll',tog,{passive:true});
  tog();
})();
